% 定义基础色 (低饱和度，护眼)
\definecolor{MorandiBlue}{RGB}{102, 129, 149}     % 雾霾蓝 (沉稳)
\definecolor{MorandiGreen}{RGB}{138, 157, 122}    % 橄榄绿 (自然)
\definecolor{MorandiYellow}{RGB}{204, 169, 114}   % 姜黄 (柔和提醒)
\definecolor{MorandiRed}{RGB}{196, 142, 142}      % 豆沙红 (重点)
\definecolor{MorandiBrown}{RGB}{166, 153, 140}    % 暖灰 (引用)

% 应用到 ElegantBook 全局变量
\definecolor{structurecolor}{named}{MorandiBlue}  % 章节/目录/页眉
\definecolor{main}{named}{MorandiGreen}           % 定义/例题环境
\definecolor{second}{named}{MorandiYellow}        % 定理环境
\definecolor{third}{named}{MorandiRed}            % 结论环境
\colorlet{coverlinecolor}{MorandiBlue}            % 封面线条

% =========================================================
%  4. 自定义文本框 (已修改为跟随莫兰迪配色)
% =========================================================

% 1. 核心思想框 (Core Box) -> 跟随主色调 (雾霾蓝)
\newtcolorbox{corebox}[1]{
  colback=structurecolor!10!white,   % 背景色：淡蓝
  colframe=structurecolor,           % 边框色：雾霾蓝
  fonttitle=\bfseries,
  title={#1},
  enhanced,
  attach boxed title to top left={yshift=-2mm, xshift=2mm},
  boxed title style={colback=structurecolor}
}

% 2. 例题框 (Example Box) -> 跟随 Main 色 (橄榄绿)
\newtcolorbox{examplebox}[1]{
  colback=main!10!white,             % 背景色：淡绿
  colframe=main,                     % 边框色：橄榄绿
  fonttitle=\bfseries,
  title={#1},
  enhanced,
  attach boxed title to top left={yshift=-2mm, xshift=2mm},
  boxed title style={colback=main}
}

% 3. 警告/陷阱框 (Alert Box) -> 跟随 Third 色 (豆沙红)
\newtcolorbox{alertbox}[1]{
